01 · the mark

The hat. The bird. The signal.

The trooper's flat-brim campaign hat — the very headgear that earned WSP airmen the nickname Smokey on CB radios in the 70s. Above the crown: a small blip and arc, equal parts circling aircraft and radio wave. Truckers called them Smokey. We just kept the name.

07 · drop into next.js

The metadata block.

Paste the JSX into app/layout.tsx as the exported metadata object. Save manifest.json at public/manifest.json and the icons under public/icons/.

// app/layout.tsx
import type { Metadata, Viewport } from 'next';

export const metadata: Metadata = {
  metadataBase: new URL('https://smokysignal.app'),
  title: { default: 'SmokySignal — Is the bird up?', template: '%s · SmokySignal' },
  description: 'Live WSP & sheriff aircraft tracker for King & Pierce County riders. Know when Smoky\u2019s watching.',
  applicationName: 'SmokySignal',
  manifest: '/manifest.json',
  themeColor: '#0b0d10',
  icons: {
    icon: [
      { url: '/icons/favicon.svg', type: 'image/svg+xml' },
      { url: '/icons/favicon-16.png', sizes: '16x16', type: 'image/png' },
      { url: '/icons/favicon-32.png', sizes: '32x32', type: 'image/png' },
      { url: '/icons/favicon-96.png', sizes: '96x96', type: 'image/png' },
    ],
    apple: [{ url: '/icons/apple-touch-icon.png', sizes: '180x180' }],
    other: [
      { rel: 'mask-icon', url: '/icons/safari-pinned-tab.svg', color: '#f5b840' },
    ],
  },
  appleWebApp: {
    capable: true,
    title: 'SmokySignal',
    statusBarStyle: 'black-translucent',
  },
  openGraph: {
    type: 'website',
    siteName: 'SmokySignal',
    title: 'Is the bird up?',
    description: 'Live aircraft tracker for King & Pierce County riders.',
    images: [{ url: '/icons/og-image.png', width: 1200, height: 630 }],
  },
  twitter: {
    card: 'summary_large_image',
    title: 'SmokySignal — Is the bird up?',
    description: 'Live aircraft tracker for King & Pierce County riders.',
    images: ['/icons/og-image.png'],
  },
};

export const viewport: Viewport = {
  themeColor: '#0b0d10',
  colorScheme: 'dark',
};
